## Common INV Tables

```sql
-- Item master
SELECT INVENTORY_ITEM_ID, SEGMENT1, DESCRIPTION
FROM APPS.MTL_SYSTEM_ITEMS_B
WHERE ORGANIZATION_ID = :org_id
  AND ENABLED_FLAG = 'Y'

-- On-hand quantities (current stock)
SELECT INVENTORY_ITEM_ID, SUBINVENTORY_CODE, SUM(TRANSACTION_QUANTITY) AS QTY
FROM APPS.MTL_ONHAND_QUANTITIES_DETAIL
WHERE ORGANIZATION_ID = :org_id
GROUP BY INVENTORY_ITEM_ID, SUBINVENTORY_CODE

-- Item categories
SELECT MIC.INVENTORY_ITEM_ID, MCS.SEGMENT1 AS CATEGORY
FROM APPS.MTL_ITEM_CATEGORIES MIC
JOIN APPS.MTL_CATEGORIES_B MCB ON MIC.CATEGORY_ID = MCB.CATEGORY_ID
JOIN APPS.MTL_CATEGORY_SETS MCS ON MIC.CATEGORY_SET_ID = MCS.CATEGORY_SET_ID
WHERE MIC.ORGANIZATION_ID = :org_id
```

## Common PO Tables

```sql
-- Purchase order headers
SELECT PO_HEADER_ID, SEGMENT1 AS PO_NUMBER, VENDOR_ID, STATUS
FROM APPS.PO_HEADERS_ALL
WHERE ORG_ID = :org_id
  AND TYPE_LOOKUP_CODE = 'STANDARD'

-- Purchase order lines
SELECT POL.PO_LINE_ID, POL.LINE_NUM, POL.ITEM_ID, POL.QUANTITY, POL.UNIT_PRICE
FROM APPS.PO_LINES_ALL POL
JOIN APPS.PO_HEADERS_ALL POH ON POL.PO_HEADER_ID = POH.PO_HEADER_ID
WHERE POH.ORG_ID = :org_id
```

## Common AP Tables

```sql
-- Invoices
SELECT INVOICE_ID, INVOICE_NUM, VENDOR_ID, INVOICE_AMOUNT, PAYMENT_STATUS_FLAG
FROM APPS.AP_INVOICES_ALL
WHERE ORG_ID = :org_id
  AND INVOICE_TYPE_LOOKUP_CODE = 'STANDARD'

-- Invoice lines
SELECT AIL.LINE_NUMBER, AIL.AMOUNT, AIL.DESCRIPTION
FROM APPS.AP_INVOICE_LINES_ALL AIL
WHERE AIL.INVOICE_ID = :invoice_id
```

## Common RCV Tables

```sql
-- Receipts
SELECT RSH.RECEIPT_NUM, RSH.RECEIPT_DATE, RSH.VENDOR_ID
FROM APPS.RCV_SHIPMENT_HEADERS RSH
WHERE RSH.RECEIPT_SOURCE_CODE = 'VENDOR'

-- Receipt transactions
SELECT RT.TRANSACTION_ID, RT.TRANSACTION_TYPE, RT.QUANTITY, RT.UNIT_OF_MEASURE
FROM APPS.RCV_TRANSACTIONS RT
WHERE RT.SHIPMENT_HEADER_ID = :shipment_header_id
```
